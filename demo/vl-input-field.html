Input field reageert op input-event

            
                <label for="input-field-met-event" block>Origineel</label>
                <input is="vl-input-field" id="input-field-met-event" block/>
                <label for="input-field-copy" block>Kopie</label>
                <input is="vl-input-field" id="input-field-copy" disabled block />

                <script>
                    customElements.whenDefined('vl-input-field').then(() => {
                        const inputField = document.querySelector('#input-field-met-event');
                        inputField.addEventListener('input', (e) => {
                            const copy = document.querySelector('#input-field-copy');
                            copy.value = inputField.value;
                        });
                    });
                </script>
